    -- Criar a tabela usuarios

CREATE TABLE usuarios (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    senha VARCHAR(255) NOT NULL,
    papel VARCHAR(10) CHECK (papel IN ('aluno', 'professor')) NOT NULL,
    telefone VARCHAR(15),
    data_nasc VARCHAR(10),
    avatar_path VARCHAR(255),
    data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL
);

CREATE TABLE quizes (
    id SERIAL PRIMARY KEY,
    titulo VARCHAR(255) NOT NULL,
    descricao TEXT NOT NULL,
    id_professor INTEGER NOT NULL REFERENCES usuarios(id),
    dificuldade VARCHAR(7) CHECK (dificuldade IN ('Fácil', 'Médio', 'Difícil')) NOT NULL,
    questoes INT NOT NULL,
    imagem_path VARCHAR(255) NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
