    -- Criar a tabela usuarios

CREATE TABLE usuarios (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    senha VARCHAR(255) NOT NULL,
    papel VARCHAR(10) CHECK (papel IN ('aluno', 'professor')) NOT NULL,
    telefone VARCHAR(15),
    data_nasc VARCHAR(10),
    avatar_path VARCHAR(255),
    data_criacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP NOT NULL,
    pontuacao_total INTEGER DEFAULT 0,
    licoes_concluidas INTEGER DEFAULT 0
);

CREATE TABLE quizes (
    id SERIAL PRIMARY KEY,
    titulo VARCHAR(255) NOT NULL,
    descricao TEXT NOT NULL,
    id_professor INTEGER NOT NULL REFERENCES usuarios(id),
    dificuldade VARCHAR(7) CHECK (dificuldade IN ('Fácil', 'Médio', 'Difícil')) NOT NULL,
    questoes INT NOT NULL,
    imagem_path VARCHAR(255) NOT NULL,
    categoria VARCHAR(10) CHECK (categoria IN ('Tecnologia', 'Ciências', 'História', 'Matemática', 'Geografia', 'Artes')) NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE questoes (
    id SERIAL PRIMARY KEY,
    id_quiz INTEGER NOT NULL REFERENCES quizes(id) ON DELETE CASCADE,
    enunciado VARCHAR(1000) NOT NULL,
    imagem_path VARCHAR(255),
    alternativa_a VARCHAR(1000) NOT NULL,
    alternativa_b VARCHAR(1000) NOT NULL,
    alternativa_c VARCHAR(1000) NOT NULL,
    alternativa_d VARCHAR(1000) NOT NULL,
    alternativa_e VARCHAR(1000) NOT NULL,
    resposta_correta VARCHAR(1) CHECK (resposta_correta IN ('A', 'B', 'C', 'D', 'E')) NOT NULL,
    explicacao VARCHAR(1000), -- Explicação opcional para a resposta
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);
